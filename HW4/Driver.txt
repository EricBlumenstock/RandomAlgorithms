using System;

using System.Linq;
using System.Text;

namespace HW4
{
    class Driver
    {
        static void Main(string[] args)
        {
            int size = Convert.ToInt32(Console.ReadLine());
            Heap heap = new Heap(size);
            for (int i = 0; i < size; i++)
            {
                heap.insert(Console.ReadLine());
                heap.print();
            }

            for (int i = 0; i < size; i++)
            {
                heap.extract();
                heap.print();
            }
           

            //heap.heapSort();
            heap.print();
            Console.ReadLine();
        }
    }
